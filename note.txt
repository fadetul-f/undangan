# inisialisai node agar bisa digunakan untuk install tailwind
- npm init -y


# install tailwind
- npm install tailwindcss @tailwindcss/cli


# build css tailwind
tambahkan di file package.json
"scripts": {
    "build-css": "tailwindcss -i ./src/style.css -o ./public/style.css --watch"
},

jalankan dengan 
- npm run build-css
kalau ini gak bisa, jalankan
- npx tailwindcss -i ./src/style.css -o ./public/style.css -w
